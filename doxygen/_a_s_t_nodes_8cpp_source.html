00072 llvm::Value *BinaryExprAST::Codegen() {
00073   // Special case '=' because we don't want to emit the LHS as an expression.
00074   if (Op == '=') {
00075     // Assignment requires the LHS to be an identifier.
00076     VariableExprAST *LHSE = llvm::dyn_cast<VariableExprAST>(LHS);
00077     if (!LHSE)
00078       return ErrorV("destination of '=' must be a variable");
00079     // Codegen the RHS.
00080     llvm::Value *Val = RHS->Codegen();
00081     if (Val == 0) return 0;
00082 
00083     // Look up the name.
00084     llvm::Value *Variable = NamedValues[LHSE->getName()];
00085     if (Variable == 0) return ErrorV("Unknown variable name");
00086 
00087     Builder.CreateStore(Val, Variable);
00088     return Val;
00089   }
00090 
00091   llvm::Value *L = LHS->Codegen();
00092   llvm::Value *R = RHS->Codegen();
00093   if (L == 0 || R == 0) return 0;
00094 
00095   switch (Op) {
00096   case '+': return Builder.CreateFAdd(L, R, "addtmp");
00097   case '-': return Builder.CreateFSub(L, R, "subtmp");
00098   case '*': return Builder.CreateFMul(L, R, "multmp");
00099   case '<':
00100             L = Builder.CreateFCmpULT(L, R, "cmptmp");
00101             // Convert bool 0/1 to double 0.0 or 1.0
00102             return Builder.CreateUIToFP(
00103               L,
00104               llvm::Type::getDoubleTy(llvm::getGlobalContext()),
00105               "booltmp");
00106   default: break;
00107   }
00108 
00109   // If it wasn't a builtin binary operator, it must be a user defined one. Emit
00110   // a call to it.
00111   llvm::Function *F = TheModule->getFunction(std::string("binary")+Op);
00112   assert(F && "binary operator not found!");
00113 
00114   llvm::Value *Ops[2] = { L, R };
00115   return Builder.CreateCall(F, Ops, "binop");
00116 }

00137 llvm::Value *IfExprAST::Codegen() {
00138   llvm::Value *CondV = Cond->Codegen();
00139   if (CondV == 0) return 0;
00140 
00141   // Convert condition to a bool by comparing equal to 0.0.
00142   CondV = Builder.CreateFCmpONE(
00143     CondV,
00144     llvm::ConstantFP::get(llvm::getGlobalContext(), llvm::APFloat(0.0)),
00145     "ifcond");
00146 
00147   llvm::Function *TheFunction = Builder.GetInsertBlock()->getParent();
00148 
00149   // Create blocks for the then and else cases.  Insert the 'then' block at the
00150   // end of the function.
00151   llvm::BasicBlock *ThenBB = llvm::BasicBlock::Create(
00152     llvm::getGlobalContext(),
00153     "then",
00154     TheFunction);
00155   llvm::BasicBlock *ElseBB = llvm::BasicBlock::Create(
00156     llvm::getGlobalContext(),
00157     "else");
00158   llvm::BasicBlock *MergeBB = llvm::BasicBlock::Create(
00159     llvm::getGlobalContext(),
00160     "ifcont");
00161 
00162   Builder.CreateCondBr(CondV, ThenBB, ElseBB);
00163 
00164   // Emit then value.
00165   Builder.SetInsertPoint(ThenBB);
00166 
00167   llvm::Value *ThenV = Then->Codegen();
00168   if (ThenV == 0) return 0;
00169 
00170   Builder.CreateBr(MergeBB);
00171   // Codegen of 'Then' can change the current block, update ThenBB for the PHI.
00172   ThenBB = Builder.GetInsertBlock();
00173 
00174   // Emit else block.
00175   TheFunction->getBasicBlockList().push_back(ElseBB);
00176   Builder.SetInsertPoint(ElseBB);
00177 
00178   llvm::Value *ElseV = Else->Codegen();
00179   if (ElseV == 0) return 0;
00180 
00181   Builder.CreateBr(MergeBB);
00182   // Codegen of 'Else' can change the current block, update ElseBB for the PHI.
00183   ElseBB = Builder.GetInsertBlock();
00184 
00185   // Emit merge block.
00186   TheFunction->getBasicBlockList().push_back(MergeBB);
00187   Builder.SetInsertPoint(MergeBB);
00188   llvm::PHINode *PN = Builder.CreatePHI(
00189     llvm::Type::getDoubleTy(llvm::getGlobalContext()),
00190     2,
00191     "iftmp");
00192 
00193   PN->addIncoming(ThenV, ThenBB);
00194   PN->addIncoming(ElseV, ElseBB);
00195   return PN;
00196 }

00198 llvm::Value *ForExprAST::Codegen() {
00199   // Output this as:
00200   //   var = alloca double
00201   //   ...
00202   //   start = startexpr
00203   //   store start -> var
00204   //   goto loop
00205   // loop:
00206   //   ...
00207   //   bodyexpr
00208   //   ...
00209   // loopend:
00210   //   step = stepexpr
00211   //   endcond = endexpr
00212   //
00213   //   curvar = load var
00214   //   nextvar = curvar + step
00215   //   store nextvar -> var
00216   //   br endcond, loop, endloop
00217   // outloop:
00218 
00219   llvm::Function *TheFunction = Builder.GetInsertBlock()->getParent();
00220 
00221   // Create an alloca for the variable in the entry block.
00222   llvm::AllocaInst *Alloca = CreateEntryBlockAlloca(TheFunction, VarName);
00223 
00224   // Emit the start code first, without 'variable' in scope.
00225   llvm::Value *StartVal = Start->Codegen();
00226   if (StartVal == 0) return 0;
00227 
00228   // Store the value into the alloca.
00229   Builder.CreateStore(StartVal, Alloca);
00230 
00231   // Make the new basic block for the loop header, inserting after current
00232   // block.
00233   llvm::BasicBlock *LoopBB = llvm::BasicBlock::Create(
00234     llvm::getGlobalContext(),
00235     "loop",
00236     TheFunction);
00237 
00238   // Insert an explicit fall through from the current block to the LoopBB.
00239   Builder.CreateBr(LoopBB);
00240 
00241   // Start insertion in LoopBB.
00242   Builder.SetInsertPoint(LoopBB);
00243 
00244   // Within the loop, the variable is defined equal to the PHI node.  If it
00245   // shadows an existing variable, we have to restore it, so save it now.
00246   llvm::AllocaInst *OldVal = NamedValues[VarName];
00247   NamedValues[VarName] = Alloca;
00248 
00249   // Emit the body of the loop.  This, like any other expr, can change the
00250   // current BB.  Note that we ignore the value computed by the body, but don't
00251   // allow an error.
00252   if (Body->Codegen() == 0)
00253     return 0;
00254 
00255   // Emit the step value.
00256   llvm::Value *StepVal;
00257   if (Step) {
00258     StepVal = Step->Codegen();
00259     if (StepVal == 0) return 0;
00260   } else {
00261     // If not specified, use 1.0.
00262     StepVal = llvm::ConstantFP::get(
00263       llvm::getGlobalContext(),
00264       llvm::APFloat(1.0));
00265   }
00266 
00267   // Compute the end condition.
00268   llvm::Value *EndCond = End->Codegen();
00269   if (EndCond == 0) return EndCond;
00270 
00271   // Reload, increment, and restore the alloca.  This handles the case where
00272   // the body of the loop mutates the variable.
00273   llvm::Value *CurVar = Builder.CreateLoad(Alloca, VarName.c_str());
00274   llvm::Value *NextVar = Builder.CreateFAdd(CurVar, StepVal, "nextvar");
00275   Builder.CreateStore(NextVar, Alloca);
00276 
00277   // Convert condition to a bool by comparing equal to 0.0.
00278   EndCond = Builder.CreateFCmpONE(
00279     EndCond,
00280     llvm::ConstantFP::get(llvm::getGlobalContext(), llvm::APFloat(0.0)),
00281     "loopcond");
00282 
00283   // Create the "after loop" block and insert it.
00284   llvm::BasicBlock *AfterBB = llvm::BasicBlock::Create(
00285     llvm::getGlobalContext(),
00286     "afterloop",
00287     TheFunction);
00288 
00289   // Insert the conditional branch into the end of LoopEndBB.
00290   Builder.CreateCondBr(EndCond, LoopBB, AfterBB);
00291 
00292   // Any new code will be inserted in AfterBB.
00293   Builder.SetInsertPoint(AfterBB);
00294 
00295   // Restore the unshadowed variable.
00296   if (OldVal)
00297     NamedValues[VarName] = OldVal;
00298   else
00299     NamedValues.erase(VarName);
00300 
00301   // for expr always returns 0.0.
00302   return llvm::Constant::getNullValue(
00303     llvm::Type::getDoubleTy(llvm::getGlobalContext()));
00304 }

00307 llvm::Value *VarExprAST::Codegen() {
00308   std::vector<llvm::AllocaInst *> OldBindings;
00309 
00310   llvm::Function *TheFunction = Builder.GetInsertBlock()->getParent();
00311 
00312   // Register all variables and emit their initializer.
00313   for (unsigned i = 0, e = VarNames.size(); i != e; ++i) {
00314     const std::string &VarName = VarNames[i].first;
00315     ExprAST *Init = VarNames[i].second;
00316 
00317     // Emit the initializer before adding the variable to scope, this prevents
00318     // the initializer from referencing the variable itself, and permits stuff
00319     // like this:
00320     //  var a = 1 in
00321     //    var a = a in ...   # refers to outer 'a'.
00322     llvm::Value *InitVal;
00323     if (Init) {
00324       InitVal = Init->Codegen();
00325       if (InitVal == 0) return 0;
00326     } else { // If not specified, use 0.0.
00327       InitVal = llvm::ConstantFP::get(llvm::getGlobalContext(),
00328                                       llvm::APFloat(0.0));
00329     }
00330 
00331     llvm::AllocaInst *Alloca = CreateEntryBlockAlloca(TheFunction, VarName);
00332     Builder.CreateStore(InitVal, Alloca);
00333 
00334     // Remember the old variable binding so that we can restore the binding when
00335     // we unrecurse.
00336     OldBindings.push_back(NamedValues[VarName]);
00337 
00338     // Remember this binding.
00339     NamedValues[VarName] = Alloca;
00340   }
00341 
00342   // Codegen the body, now that all vars are in scope.
00343   llvm::Value *BodyVal = Body->Codegen();
00344   if (BodyVal == 0) return 0;
00345 
00346   // Pop all our variables from scope.
00347   for (unsigned i = 0, e = VarNames.size(); i != e; ++i)
00348     NamedValues[VarNames[i].first] = OldBindings[i];
00349 
00350   // Return the body computation.
00351   return BodyVal;
00352 }

00355 llvm::Function *PrototypeAST::Codegen() {
00356   // Make the function type:  double(double,double) etc.
00357   std::vector<llvm::Type*> Doubles(
00358     Args.size(),
00359     llvm::Type::getDoubleTy(llvm::getGlobalContext()));
00360   llvm::FunctionType *FT = llvm::FunctionType::get(
00361     llvm::Type::getDoubleTy(llvm::getGlobalContext()),
00362     Doubles,
00363     false);
00364 
00365   llvm::Function *F = llvm::Function::Create(
00366     FT,
00367     llvm::Function::ExternalLinkage,
00368     Name,
00369     TheModule);
00370 
00371   // If F conflicted, there was already something named 'Name'.  If it has a
00372   // body, don't allow redefinition or reextern.
00373   if (F->getName() != Name) {
00374     // Delete the one we just made and get the existing one.
00375     F->eraseFromParent();
00376     F = TheModule->getFunction(Name);
00377 
00378     // If F already has a body, reject this.
00379     if (!F->empty()) {
00380       ErrorF("redefinition of function");
00381       return 0;
00382     }
00383 
00384     // If F took a different number of args, reject.
00385     if (F->arg_size() != Args.size()) {
00386       ErrorF("redefinition of function with different # args");
00387       return 0;
00388     }
00389   }
00390 
00391   // Set names for all arguments.
00392   unsigned Idx = 0;
00393   for (llvm::Function::arg_iterator AI = F->arg_begin(); Idx != Args.size();
00394        ++AI, ++Idx)
00395     AI->setName(Args[Idx]);
00396 
00397   return F;
00398 }

00428       Proto->getBinaryPrecedence();

00432     llvm::getGlobalContext(),
00433     "entry",

00438   Proto->CreateArgumentAllocas(TheFunction);

00440   if (llvm::Value *RetVal = Body->Codegen()) {
00441     // Finish off the function.

00447     //----------------------
