00237 ExprAST *Parser::ParseBinOpRHS(int ExprPrec, ExprAST *LHS) {
00238   // If this is a binop, find its precedence.
00239   while (1) {
00240     int TokPrec = Tok.GetTokPrecedence();
00241 
00242     // If this is a binop that binds at least as tightly as the current binop,
00243     // consume it, otherwise we are done.
00244     if (TokPrec < ExprPrec)
00245       return LHS;
00246 
00247     // Okay, we know this is a binop.
00248     int BinOp = Tok.Kind;
00249     GetNextToken();  // eat binop
00250 
00251     // Parse the unary expression after the binary operator.
00252     ExprAST *RHS = ParseUnary();
00253     if (!RHS) return 0;
00254 
00255     // If BinOp binds less tightly with RHS than the operator after RHS, let
00256     // the pending operator take RHS as its LHS.
00257     int NextPrec = Tok.GetTokPrecedence();
00258     if (TokPrec < NextPrec) {
00259       RHS = ParseBinOpRHS(TokPrec+1, RHS);
00260       if (RHS == 0) return 0;
00261     }
00262 
00263     // Merge LHS/RHS.
00264     LHS = new BinaryExprAST(BinOp, LHS, RHS);
00265   }
00266 }

00285

00287 
00288   unsigned Kind = 0; // 0 = identifier, 1 = unary, 2 = binary.
00289   unsigned BinaryPrecedence = 30;
00290 
00291   switch (Tok.Kind) {
00292   default:
00293     return ErrorP("Expected function name in prototype");
00294   case tok::tok_identifier:
00295     FnName = Tok.IdentifierStr;
00296     Kind = 0;
00297     GetNextToken();
00298     break;
00299   case tok::tok_unary:
00300     GetNextToken();
00301     if (!isascii(Tok.Kind))
00302       return ErrorP("Expected unary operator");
00303     FnName = "unary";
00304     FnName += (char)Tok.Kind;
00305     Kind = 1;
00306     GetNextToken();
00307     break;
00308   case tok::tok_binary:
00309     GetNextToken();
00310     if (!isascii(Tok.Kind))
00311       return ErrorP("Expected binary operator");
00312     FnName = "binary";
00313     FnName += (char)Tok.Kind;
00314     Kind = 2;
00315     GetNextToken();
00316 
00317     // Read the precedence if present.
00318     if (Tok.Kind == tok::tok_number) {
00319       if (Tok.NumVal < 1 || Tok.NumVal > 100)
00320         return ErrorP("Invalid precedecnce: must be 1..100");
00321       BinaryPrecedence = (unsigned)(Tok.NumVal);
00322       GetNextToken();
00323     }
00324     break;
00325   }
00326 
00327   if (Tok.Kind != '(')
00328     return ErrorP("Expected '(' in prototype");
00329 
00330   std::vector<std::string> ArgNames;
00331   while (GetNextToken() == tok::tok_identifier)
00332     ArgNames.push_back(Tok.IdentifierStr);
00333   if (Tok.Kind != ')')
00334     return ErrorP("Expected ')' in prototype");
00335 
00336   // success.
00337   GetNextToken();  // eat ')'.
00338 
00339   // Verify right number of names for operator.
00340   if (Kind && ArgNames.size() != Kind)
00341     return ErrorP("Invalid number of operands for operator");
00342 
00343   return new PrototypeAST(FnName, ArgNames, Kind != 0, BinaryPrecedence);
00344 }
