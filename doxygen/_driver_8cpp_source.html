00067 
00068   //-----------------------------------------------------
00069   // Create the JIT.  This takes ownership of the module.
00070   std::string ErrStr;
00071   klang::TheExecutionEngine =
00072     llvm::EngineBuilder(klang::TheModule).setErrorStr(&ErrStr).create();
00073   if (!klang::TheExecutionEngine) {
00074     fprintf(stderr, "Could not create ExecutionEngine: %s\n", ErrStr.c_str());
00075     exit(1);

00077 
00078   llvm::FunctionPassManager OurFPM(klang::TheModule);
00079 
00080   // Set up the optimizer pipeline.  Start with registering info about how the
00081   // target lays out data structures.
00082   OurFPM.add(new llvm::TargetData(*klang::TheExecutionEngine->getTargetData()));
00083   // Provide basic AliasAnalysis support for GVN.
00084   OurFPM.add(llvm::createBasicAliasAnalysisPass());
00085   // Promote allocas to registers.
00086   OurFPM.add(llvm::createPromoteMemoryToRegisterPass());
00087   // Do simple "peephole" optimizations and bit-twiddling optzns.
00088   OurFPM.add(llvm::createInstructionCombiningPass());
00089   // Reassociate expressions.
00090   OurFPM.add(llvm::createReassociatePass());
00091   // Eliminate Common SubExpressions.
00092   OurFPM.add(llvm::createGVNPass());
00093   // Simplify the control flow graph (deleting unreachable blocks, etc).
00094   OurFPM.add(llvm::createCFGSimplificationPass());
00095 
00096   OurFPM.doInitialization();
00097 
00098   // Set the global so the code gen can use this.
00099   klang::TheFPM = &OurFPM;
