00068 int main(int argc, char* const argv[]) {
00069 
00070   llvm::cl::ParseCommandLineOptions(argc, argv);
00071 
00072   llvm::OwningPtr<llvm::MemoryBuffer> Buf;
00073 
00074   if (llvm::MemoryBuffer::getFileOrSTDIN(InputFilename, Buf))
00075     return 1;
00076 
00077   llvm::InitializeNativeTarget();
00078   llvm::LLVMContext &Context = llvm::getGlobalContext();
00079 
00080   klang::Lexer myLexer(Buf->getBuffer());
00081   klang::Parser myParser(myLexer);
00082 
00083   // Install standard binary operators.
00084   // 1 is lowest precedence.
00085   klang::Token::BinopPrecedence['='] = 2;
00086   klang::Token::BinopPrecedence['<'] = 10;
00087   klang::Token::BinopPrecedence['+'] = 20;
00088   klang::Token::BinopPrecedence['-'] = 20;
00089   klang::Token::BinopPrecedence['*'] = 40;  // highest.
00090 
00091   // Make the module, which holds all the code.
00092   klang::TheModule = new llvm::Module("my cool jit", Context);
00093 
00094   //-----------------------------------------------------
00095   // Create the JIT.  This takes ownership of the module.
00096   std::string ErrStr;
00097   klang::TheExecutionEngine =
00098     llvm::EngineBuilder(klang::TheModule).setErrorStr(&ErrStr).create();
00099   if (!klang::TheExecutionEngine) {
00100     llvm::errs() << "Could not create ExecutionEngine: " << ErrStr.c_str()
00101       << "\n";
00102     exit(1);
00103   }
00104 
00105   llvm::FunctionPassManager OurFPM(klang::TheModule);
00106 
00107   // Set up the optimizer pipeline.  Start with registering info about how the
00108   // target lays out data structures.
00109   OurFPM.add(new llvm::DataLayout(*klang::TheExecutionEngine->getDataLayout()));
00110   // Provide basic AliasAnalysis support for GVN.
00111   OurFPM.add(llvm::createBasicAliasAnalysisPass());
00112   // Promote allocas to registers.
00113   OurFPM.add(llvm::createPromoteMemoryToRegisterPass());
00114   // Do simple "peephole" optimizations and bit-twiddling optzns.
00115   OurFPM.add(llvm::createInstructionCombiningPass());
00116   // Reassociate expressions.
00117   OurFPM.add(llvm::createReassociatePass());
00118   // Eliminate Common SubExpressions.
00119   OurFPM.add(llvm::createGVNPass());
00120   // Simplify the control flow graph (deleting unreachable blocks, etc).
00121   OurFPM.add(llvm::createCFGSimplificationPass());
00122 
00123   OurFPM.doInitialization();
00124 
00125   // Set the global so the code gen can use this.
00126   klang::TheFPM = &OurFPM;
00127   //-----------------------------------------------------
00128 
00129   // Run the main "interpreter loop" now.
00130   myParser.Go();
00131 
00132   klang::TheFPM = 0;
00133 
00134   // Print out all of the generated code.
00135   //FIXME
00136   //IR dumping will be done via a new frontendaction emit-llvm
00137   //klang::TheModule->dump();
00138 
00139   // Calls an unused function just not to lose it in the final binary
00140   // Without this call klangBuiltin.a is just ignored during linking
00141   putchard('\n');
00142 
00143   return 0;
00144 }
