00179     std::string Name;
00180     std::vector<std::string> Args;
00181     bool isOperator;
00182     unsigned Precedence;  // Precedence if a binary op.
00183   public:
00184     PrototypeAST(const std::string &name, const std::vector<std::string> &args,
00185                  bool isoperator = false, unsigned prec = 0)
00186       : Name(name), Args(args), isOperator(isoperator), Precedence(prec) {}
00187 
00188     bool isUnaryOp() const { return isOperator && Args.size() == 1; }
00189     bool isBinaryOp() const { return isOperator && Args.size() == 2; }
00190 
00191     char getOperatorName() const {
00192       assert(isUnaryOp() || isBinaryOp());
00193       return Name[Name.size()-1];
00194     }
00195 
00196     unsigned getBinaryPrecedence() const { return Precedence; }
